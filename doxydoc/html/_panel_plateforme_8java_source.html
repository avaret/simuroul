220  private BufferedImage reDessineCarteDansBuffer( boolean avecAntialiasing ) {
221 
222  BufferedImage image = new BufferedImage(getWidth(), getHeight(), BufferedImage.TYPE_INT_ARGB);
223 
224  if(_aeroport != null)
225  {
226  Graphics2D g2 = image.createGraphics();
227  if( avecAntialiasing )
228  {
229  g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING,
230  RenderingHints.VALUE_ANTIALIAS_ON);
231  g2.setRenderingHint(RenderingHints.KEY_RENDERING,
232  RenderingHints.VALUE_RENDER_QUALITY);
233  }
234  g2.setClip(0,0, getWidth(), getHeight());
235  _drawer.dessineAeroport(_aeroport, g2, _echelle.getAffineTransform());
236  g2.dispose();
237  }
