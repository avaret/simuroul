211  private class KeyDispatcher implements KeyEventDispatcher {
212  @Override
213  public boolean dispatchKeyEvent(KeyEvent e) {
214  if (e.getID() == KeyEvent.KEY_PRESSED ) {
215  if(e.getKeyCode() == KeyEvent.VK_RIGHT) {
216  _controleur.setInstant(_controleur.getInstantCourant()+InstantFabrique._pasEntreInstant);
217  return true;
218  }else if (e.getKeyCode() == KeyEvent.VK_LEFT) {
219  _controleur.setInstant(_controleur.getInstantCourant()-InstantFabrique._pasEntreInstant);
220  return true;
221  }else if (e.getKeyCode() == KeyEvent.VK_SPACE) {
222  if(_controleur.isTraficRunning())
223  _controleur.stopTrafic();
224  else
225  _controleur.runTrafic();
226 
227  return true;
228  }else if (e.getKeyCode() == KeyEvent.VK_M) {
229  JOptionPane.showMessageDialog(null, "Tips memoire :" + LibereMemoire.controleMemoire() + " Mo", "" , JOptionPane.INFORMATION_MESSAGE);
230  return true;
231  }
232 
233  return false;
234  }
235  else
236  return false;
237  }
